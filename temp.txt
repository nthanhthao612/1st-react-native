async componentDidMount() {
        const info = await AsyncStorage.getItem('lastrecordedhealthCare');
        if (!info) {
            const { data } = await axios.
                get("http://192.168.1.218:7000/api/healthcare/getfinal");
            await AsyncStorage.setItem('healthCare', JSON.stringify(data));
            let {lastRecorded} = data;
            await AsyncStorage.setItem('lastrecordedhealthCare', JSON.stringify(lastRecorded));
            delete lastRecorded._id;
            this.setState(state => {
                return {
                    Date: lastRecorded.Date
                }
            });
            // delete lastRecorded._id;
            delete lastRecorded.Date;
            this.setState(state => {
                return {
                    ListItem: lastRecorded
                }
            });
        }else{
            let parseInfo = JSON.parse(info);
            delete parseInfo._id;
            this.setState(state => {
                return {
                    Date: parseInfo.Date
                }
            });
            delete parseInfo._id;
            delete parseInfo.Date;
            this.setState(state => {
                return {
                    ListItem: parseInfo
                }
            });
        }
    }
